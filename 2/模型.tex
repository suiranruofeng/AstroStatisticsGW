	P(d[Fe/H],[Fe/H]_0|[Fe/H], r_0, \sigma^2[Fe/H],r_{err})\\
	\propto \prod_i^nP([Fe/H]_i|\sigma^2[Fe/H], \mu[Fe/H]_i)P(r_0|r, r_{err})\cdot \\

\begin{equation}

	P([Fe/H]_i|\sigma^2[Fe/H]， \mu[Fe/H]_i)\sim \mathcal{N}(\sigma^2[Fe/H], \mu[Fe/H]_i)
\end{equation}
